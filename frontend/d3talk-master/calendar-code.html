// generates path data for month t0
function monthPath(t0) {
  var t1 = lastDayOfMonth(t0),
      d0 = dayInWeek(t0), // %w
      d1 = dayInWeek(t1),
      w0 = weekInYear(t0), // %U
      w1 = weekInYear(t1);
  return "M" + (w0 + 1) * z + "," + (d0 + 0) * z
       + "H" + (w0 + 0) * z + "V" + (     7) * z
       + "H" + (w1 + 0) * z + "V" + (d1 + 1) * z
       + "H" + (w1 + 1) * z + "V" + (     0) * z
       + "H" + (w0 + 1) * z + "Z";
}
