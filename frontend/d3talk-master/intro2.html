// podemos calcular los parametros dependiendo
// de los datos (d es un dato, i indice)
d3.selectAll("p").style("color", function(d, i) {
  return i % 2 ? "#fff" : "#eee";
});

// enlazar datos
d3.selectAll("p")
  .data([4, 8, 15, 16, 23, 42])
  .style("font-size", function(d) { 
      return d + "px"; 
  });
